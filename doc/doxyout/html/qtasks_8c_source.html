161 
162  if ( NULL != Task ) {

166  qOS_Set_TaskFlags( Task, QTASK_BIT_ENABLED, (qBool_t)s );

304  qTask_Flag_t flagsToCheck,

306  const qBool_t checkForAll )
307 {
308  qBool_t retValue = qFalse;
309 
310  if ( NULL != Task ) {
311  qTask_Flag_t cEventBits = Task->qPrivate.flags & QTASK_EVENT_FLAGS_MASK;
312 
313  flagsToCheck &= QTASK_EVENT_FLAGS_MASK;
314  if ( qFalse == checkForAll ) {
315  if ( (qTask_Flag_t)0 != ( cEventBits & flagsToCheck ) ) {
316  retValue = qTrue;
317  }
318  }
319  else {
320  if ( ( cEventBits & flagsToCheck ) == flagsToCheck ) {
321  retValue = qTrue;
322  }
323  }
324  if ( ( qTrue == retValue ) && ( qTrue == clearOnExit ) ) {
325  Task->qPrivate.flags &= ~flagsToCheck;
326  }
327  }
