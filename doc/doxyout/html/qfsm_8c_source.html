196 static void qStateMachine_Transition( qSM_t *m,
197  qSM_State_t *target,
198  const qSM_TransitionHistoryMode_t mHistory )
199 {
200  /*
201  When a transition its performed, all exit actions must precede any actions
202  associated with the transition, which must precede any entry actions
203  associated with the newly entered state(s). To discover which exit actions
204  to execute, we first find the Least Common Ancestor (LCA) of the source and
205  target states.
206  */
207  qStateMachine_ExitUpToLCA( m, qStateMachine_LevelsToLCA( m, target ) );
208  /*then, handle the required history mode*/
209  if ( qSM_TRANSITION_NO_HISTORY == mHistory ) {
210  /*just restore the default transition*/
211  target->qPrivate.lastRunningChild = target->qPrivate.initState;
212  }
213  else if ( qSM_TRANSITION_SHALLOW_HISTORY == mHistory ) {
214  if ( NULL != target->qPrivate.lastRunningChild ) {
215  qSM_State_t * lrc = target->qPrivate.lastRunningChild;
216  /*restore the default transition in the last running nested state*/
217  lrc->qPrivate.lastRunningChild = lrc->qPrivate.initState;
218  }
219  }
220  else {
221  /*
222  nothing to do : qSM_TRANSITION_DEEP_HISTORY handled by default all deep
223  history its preserved in the "lastRunningChild" attribute, so there is
224  no need to change it. qStateMachine_StateOnStart will assign this by
225  default.
226  */
227  }
228  m->qPrivate.next = target; /*notify SM that there was indeed a transition*/
229 }

231 static qSM_LCA_t qStateMachine_LevelsToLCA( qSM_t * const m,
232  qSM_State_t * const target )
233 {
234  qSM_LCA_t xLca = 0u;
235 
236  /*
237  To discover which exit actions to execute, it is necessary to find the
238  LCA(LeastCommon Ancestor) of the source and target states.
239  */
240  if ( m->qPrivate.source == target ) {
241  xLca = 1u; /*recursive transition, only a level needs to be performed*/
242  }
243  else {
244  qSM_State_t *s, *t;
245  qBool_t xBreak = qFalse; /*to be in compliance with MISRAC2012-Rule-15.5*/
246  qSM_LCA_t n = 0u;
247 
248  for ( s = m->qPrivate.source ;
249  ( NULL != s ) && ( qFalse == xBreak ) ;
250  s = s->qPrivate.parent ) {
251  for ( t = target ; NULL != t ; t = t->qPrivate.parent ) {
252  if ( s == t ) {
253  xLca = n;
254  xBreak = qTrue;
255  break;
256  }
257  }
258  ++n;
259  }
260  }
261 
262  return xLca; /*return # of levels from the current state to the LCA.*/
263 }

265 static void qStateMachine_ExitUpToLCA( qSM_t * const m,
266  qSM_LCA_t lca )
267 {
268  qSM_State_t *s = m->qPrivate.current;
269 
270  /*exit current states*/
271  while ( s != m->qPrivate.source ) {
272  s = qStateMachine_StateOnExit( m, s );
273  }
274  /*exit all superstates up to LCA*/
275  while ( 0u != lca-- ) {
276  s = qStateMachine_StateOnExit( m, s );
277  }
278  m->qPrivate.current = s;
279 }

414 static void qStateMachine_TracePathAndRetraceEntry( qSM_t * const m,
415  qSM_State_t **trace )
416 {
417  qSM_State_t *s;
418  /*
419  Entry actions must be executed in order form the least deeply nested to the
420  most deeply nested state. This is opposite to the normal navigability of the
421  module data-structure design. So this is solved by first recording the entry
422  path from the LCA to the target, then playing it backwards. with execution
423  of entry actions.
424  qStateMachine_TracePathAndRetraceEntry and qStateMachine_TraceOnStart are
425  used to handle this after the transition perform all the required exit
426  actions.
427  */
428  *trace = NULL;
429  for ( s = m->qPrivate.next ; s != m->qPrivate.current ; s = s->qPrivate.parent ) {
430  *(++trace) = s; /* trace path to target */
431  }
432  while ( NULL != ( s = *trace-- ) ) { /* retrace entry from LCA */
433  qStateMachine_StateOnEntry( m, s );
434  }
435  m->qPrivate.current = m->qPrivate.next;
436  m->qPrivate.next = NULL;
437 }

447 static qSM_Signal_t qStateMachine_CheckForSignals( qSM_t * const m,
448  const qSM_Signal_t sig )
449 {
450  qSM_Signal_t xSignal = sig;
451 
452  /*
453  initially checks for time signals if the state machine has a
454  timeout-specification object installed. If any timeout expires, the signal
455  is put on the available recipient.
456  */
457  if ( NULL != m->qPrivate.timeSpec ) {
458  qStateMachine_TimeoutCheckSignals( m );
459  /*use the available recipient: signal-queue or the EV*/
460  }
461  /*
462  If the state machine has the signal-queue installed, the signal that is in
463  front of the queue is obtained. In case of there is no signal-queue or the
464  queue its empty, the input argument and the exclusion variable are verified.
465  */
466  #if ( Q_QUEUES == 1 )
467  if ( NULL != m->qPrivate.sQueue ) {
468  qSM_Signal_t receivedSignal;
469 
470  if ( qTrue == qQueue_Receive( m->qPrivate.sQueue, &receivedSignal ) ) {
471  xSignal = receivedSignal;
472  }
473  }
474  #endif
475  /*A signal coming from the signal-queue has the higher precedence.*/
476  if ( ( QSM_SIGNAL_NONE == xSignal.id ) && ( QSM_SIGNAL_NONE != m->qPrivate.signalNot.id ) ) {
477  xSignal = m->qPrivate.signalNot; /*exclusion variable*/
478  m->qPrivate.signalNot.id = QSM_SIGNAL_NONE;
479  m->qPrivate.signalNot.sigData = NULL;
480  }
481 
482  return xSignal;
483 }

485 qBool_t qStateMachine_Run( qSM_t * const m,
486  qSM_Signal_t sig )
487 {
488  qBool_t retValue = qFalse;
489 
490  if ( NULL != m ) {
491  qSM_State_t *entryPath[ Q_FSM_MAX_NEST_DEPTH ];
492  qSM_State_t *s;
493 
494  sig = qStateMachine_CheckForSignals( m, sig );
495 
496  /*Enter here only once to start the top state*/
497  if ( NULL == m->qPrivate.current ) {
498  m->qPrivate.current = &m->qPrivate.top;
499  m->qPrivate.next = NULL;
500  qStateMachine_StateOnEntry( m, m->qPrivate.current );
501  qStateMachine_TraceOnStart( m, entryPath );
502  }
503  /*evaluate the hierarchy until the signal is handled*/
504  for ( s = m->qPrivate.current ; NULL != s ; s = s->qPrivate.parent ) {
505  m->qPrivate.source = s; /* level of outermost event handler */
506  if ( qSM_STATUS_SIGNAL_HANDLED == qStateMachine_StateOnSignal( m, s, sig ) ) {
507  if ( NULL != m->qPrivate.next ) { /* state transition taken? */
508  /*
509  execute entry/start actions in the rest of the hierarchy
510  after transition
511  */
512  qStateMachine_TracePathAndRetraceEntry( m, entryPath );
513  qStateMachine_TraceOnStart( m, entryPath );
514  }
515  retValue = qTrue;
516  break;/* signal processed */
517  }
518  }
519  }
520 
521  return retValue;
522 }

744 static void qStateMachine_TimeoutPerformSpecifiedActions( qSM_t * const m,
745  qSM_State_t *current,
746  const qSM_SigId_t sig )
747 {
748  qSM_TimeoutStateDefinition_t *tbl = current->qPrivate.tdef;
749  size_t n = current->qPrivate.nTm;
750 
751  if ( ( n > 0u ) && ( NULL != tbl ) ) {
752  size_t i;
753  qSM_TimeoutSpecOptions_t setCheck, resetCheck;
754 
755  if ( QSM_SIGNAL_ENTRY == sig ) {
756  setCheck = QSM_TSOPT_SET_ENTRY;
757  resetCheck = QSM_TSOPT_RST_ENTRY;
758  }
759  else {
760  setCheck = QSM_TSOPT_SET_EXIT;
761  resetCheck = QSM_TSOPT_RST_EXIT;
762  }
763  for ( i = 0u ; i < n ; ++i ) { /*loop table */
764  qSM_TimeoutSpecOptions_t opt = tbl[ i ].options;
765  qIndex_t index = (qIndex_t)( opt & QSM_TSOPT_INDEX_MASK );
766  /*state match and index is valid?*/
767  if ( index < (qIndex_t)Q_FSM_MAX_TIMEOUTS ) {
768  qSTimer_t *tmr = &m->qPrivate.timeSpec->builtin_timeout[ index ];
769  qTime_t tValue = tbl[ i ].xTimeout;
770 
771  if ( 0uL != ( opt & setCheck ) ) {
772  if ( 0uL == ( opt & QSM_TSOPT_KEEP_IF_SET ) ) {
773  (void)qSTimer_Set( tmr, tValue );
774  }
775  if ( 0uL != ( opt & QSM_TSOPT_PERIODIC ) ) {
776  qFLM_BitSet( m->qPrivate.timeSpec->isPeriodic,
777  index,
778  qUINT32_t );
779  }
780  else {
781  qFLM_BitClear( m->qPrivate.timeSpec->isPeriodic,
782  index,
783  qUINT32_t );
784  }
785  }
786  if ( 0uL != ( opt & resetCheck ) ) {
787  (void)qSTimer_Disarm( tmr );
788  }
789  }
790  }
791  }
